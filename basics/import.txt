package basics

import (
	"fmt"
	foo "net/http"
)

func main(){
	 fmt.Println("Hello,Go standard library");

	 resp,err :=foo.Get("http://jsonplaceholder.typicode.com/posts/1")
	 if err !=nil {
			fmt.Println("Error:",err)
			return
	 }
	 defer resp.Body.Close()

	 fmt.Println("Http response status",resp.Status)
}
